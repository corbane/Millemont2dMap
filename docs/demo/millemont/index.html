function init (objElement)
{
    var map = new ImageMap.Map2d (objElement.contentDocument.querySelector ("svg")),
        info = document.getElementById ("info")

    for( var region of map.regions )
    {
        region.hSelect.add (region => { info.innerHTML = region.id })
        region.hUnselect.add (region => { info.innerHTML = "<br/>" })
    }
}
